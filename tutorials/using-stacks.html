2) Vertical layout with VStack

VStack(alignment: .leading, spacing: 12) {
    Text("Daily Standup")
        .font(.headline)
    Text("15 minutes • Design")
        .foregroundStyle(.secondary)
    Button("Start Meeting") { /* action */ }
}
.padding()

Use alignment and spacing. Add .padding() for insets. :contentReference[oaicite:3]{index=3}

3) Horizontal layout with HStack

HStack(spacing: 16) {
    Image(systemName: "clock")
    Text("Elapsed 05:21")
    Spacer()
    Text("Remaining 09:39")
}

Spacer() pushes siblings apart. :contentReference[oaicite:4]{index=4}

4) Overlay with ZStack

ZStack {
    Circle().strokeBorder(.blue.opacity(0.2), lineWidth: 10)
    Text("Timer").font(.caption)
}

ZStack layers views front‑to‑back. :contentReference[oaicite:5]{index=5}

5) Separators & spacing

VStack {
    Divider()
    HStack { Text("Speaker 1"); Spacer(); Text("00:45") }
    HStack { Text("Speaker 2"); Spacer(); Text("01:12") }
}

Divider() draws a thin rule; combine with Spacer() for flexible gaps. :contentReference[oaicite:6]{index=6}

6) Alignment refinements

VStack(alignment: .leading) {
    Text("Agenda").font(.headline)
    HStack(alignment: .firstTextBaseline) {
        Text("Design"); Text("•"); Text("Decisions")
    }
}

Choose stack alignments or use alignment guides for cross‑stack alignment when needed. :contentReference[oaicite:7]{index=7}

Android (Jetpack Compose)

1) Project & Preview

// MainActivity.kt
@Composable
fun Content() {
    Text("Scrumdinger", style = MaterialTheme.typography.titleLarge, modifier = Modifier.padding(16.dp))
}

Compose preview mirrors SwiftUI’s quick iteration.

2) Column (VStack)

@Composable
fun HeaderSection() {
    Column(
        modifier = Modifier.padding(16.dp),
        verticalArrangement = Arrangement.spacedBy(12.dp),
        horizontalAlignment = Alignment.Start
    ) {
        Text("Daily Standup", style = MaterialTheme.typography.titleMedium)
        Text("15 minutes • Design", color = MaterialTheme.colorScheme.onSurfaceVariant)
        Button(onClick = { /* TODO */ }) { Text("Start Meeting") }
    }
}

3) Row (HStack) with Spacer

@Composable
fun TimeRow() {
    Row(
        modifier = Modifier.fillMaxWidth(),
        horizontalArrangement = Arrangement.spacedBy(16.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        Icon(Icons.Default.AccessTime, contentDescription = null)
        Text("Elapsed 05:21")
        Spacer(Modifier.weight(1f))
        Text("Remaining 09:39")
    }
}

4) Box (ZStack)

@Composable
fun TimerOverlay() {
    Box(contentAlignment = Alignment.Center, modifier = Modifier.size(120.dp)) {
        Canvas(modifier = Modifier.matchParentSize()) {
            drawCircle(color = Color(0x33007AFF), style = Stroke(width = 10f))
        }
        Text("Timer", style = MaterialTheme.typography.labelSmall)
    }
}

5) Divider & spacing

@Composable
fun Speakers() {
    Column {
        Divider()
        Row(Modifier.fillMaxWidth().padding(horizontal = 8.dp)) {
            Text("Speaker 1"); Spacer(Modifier.weight(1f)); Text("00:45")
        }
        Row(Modifier.fillMaxWidth().padding(horizontal = 8.dp)) {
            Text("Speaker 2"); Spacer(Modifier.weight(1f)); Text("01:12")
        }
    }
}

6) Alignment details

@Composable
fun Agenda() {
    Column(horizontalAlignment = Alignment.Start) {
        Text("Agenda", style = MaterialTheme.typography.titleSmall)
        Row(verticalAlignment = Alignment.CenterVertically) {
            Text("Design"); Text(" • "); Text("Decisions")
        }
    }
}

Flutter (Dart)

1) Project & Hot Reload

// lib/main.dart
import 'package:flutter/material.dart';

void main() => runApp(const MyApp());

class MyApp extends StatelessWidget {
  const MyApp({super.key});
  @override
  Widget build(BuildContext context) {
    return MaterialApp(home: Scaffold(body: Content()));
  }
}

class Content extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return const Padding(
      padding: EdgeInsets.all(16),
      child: Text('Scrumdinger', style: TextStyle(fontSize: 22)),
    );
  }
}

2) Column (VStack)

Widget headerSection() {
  return Padding(
    padding: const EdgeInsets.all(16),
    child: Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      mainAxisSize: MainAxisSize.min,
      children: [
        const Text('Daily Standup', style: TextStyle(fontSize: 20, fontWeight: FontWeight.w600)),
        const SizedBox(height: 12),
        Text('15 minutes • Design', style: TextStyle(color: Colors.grey[700])),
        const SizedBox(height: 12),
        FilledButton(onPressed: () {}, child: const Text('Start Meeting')),
      ],
    ),
  );
}

3) Row (HStack) with Spacer

Widget timeRow() {
  return Row(
    children: const [
      Icon(Icons.access_time),
      SizedBox(width: 16),
      Text('Elapsed 05:21'),
      Spacer(),
      Text('Remaining 09:39'),
    ],
  );
}

4) Stack (ZStack)

Widget timerOverlay() {
  return SizedBox(
    width: 120, height: 120,
    child: Stack(alignment: Alignment.center, children: [
      // Simple ring using a Container + decoration; for custom draw use CustomPaint
      Container(
        decoration: BoxDecoration(shape: BoxShape.circle, border: Border.all(color: const Color(0x33007AFF), width: 10)),
      ),
      const Text('Timer', style: TextStyle(fontSize: 12)),
    ]),
  );
}

5) Divider & spacing

Widget speakers() {
  return Column(
    children: const [
      Divider(),
      Padding(
        padding: EdgeInsets.symmetric(horizontal: 8),
        child: Row(children: [Text('Speaker 1'), Spacer(), Text('00:45')]),
      ),
      Padding(
        padding: EdgeInsets.symmetric(horizontal: 8),
        child: Row(children: [Text('Speaker 2'), Spacer(), Text('01:12')]),
      ),
    ],
  );
}

6) Alignment details

Widget agenda() {
  return Column(
    crossAxisAlignment: CrossAxisAlignment.start,
    children: const [
      Text('Agenda', style: TextStyle(fontWeight: FontWeight.w600)),
      Row(children: [Text('Design'), Text(' • '), Text('Decisions')]),
    ],
  );
}
